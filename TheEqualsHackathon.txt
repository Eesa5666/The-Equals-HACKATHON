    <script>
        function goToFinancePage() {
            document.getElementById('personalInfoContainer').style.display = 'none';
            document.getElementById('financeContainer').style.display = 'block';
        }

        async function aiGenerateInvestmentPlan() {
            const income = parseFloat(document.getElementById('income').value);
            const expenses = parseFloat(document.getElementById('expenses').value);
            const savings = parseFloat(document.getElementById('savings').value);
            const risk = parseInt(document.getElementById('risk').value);
            
            if (isNaN(income) || isNaN(expenses) || isNaN(savings) || isNaN(risk)) {
                document.getElementById('output').innerHTML = 'Please enter valid numbers for all fields.';
                return;
            }
            
            const investableAmount = income - expenses;
            let strategy;
            
            if (risk <= 3) {
                strategy = "Consider low-risk investments like bonds and high-yield savings accounts.";
            } else if (risk <= 7) {
                strategy = "A balanced portfolio with mutual funds and ETFs may suit your needs.";
            } else {
                strategy = "You can explore high-risk investments like stocks and crypto for higher returns.";
            }
            
            document.getElementById('output').innerHTML = `
                <strong>Investment Plan:</strong><br>
                Monthly Investable Amount: $${investableAmount.toFixed(2)}<br>
                Suggested Strategy: ${strategy}
            `;
        }
        
        async function aiFetchMarketNews() {
            const news = [
                "Stock market rises amid economic recovery hopes.",
                "Tech stocks lead gains as investors bet on innovation.",
                "Oil prices surge as demand outlook improves.",
                "AI-driven automation continues to shape the financial sector."
            ];
            
            let newsHTML = '<strong>Market News:</strong><br>';
            news.forEach(item => {
                newsHTML += <div class='summary'>${item}</div>;
            });
            
            document.getElementById('marketNews').innerHTML = newsHTML;
        }
    </script>
